  DATA : lt_qc_h TYPE TABLE OF ztc_qc_h,
         ls_qc_h LIKE LINE OF lt_qc_h.
  DATA : lv_erdat TYPE ztc_qc_h-erdat.

  SELECT qc_doc_num  erdat
    FROM ztc_qc_h
    INTO CORRESPONDING FIELDS OF TABLE lt_qc_h
  .

  SELECT wh_id, sku_id,SUM( good_qty ) AS real_qty,
    qc_doc_num AS pp_doc_num
     FROM ztc_qc_i
    INTO CORRESPONDING FIELDS OF  @gs_ppend
    GROUP BY wh_id, sku_id, qc_doc_num .

    READ TABLE lt_qc_h INTO ls_qc_h TRANSPORTING erdat WITH KEY qc_doc_num = gs_ppend-pp_doc_num .
    MOVE ls_qc_h-erdat TO gs_ppend-end_date.
    CLEAR : ls_qc_h.

    APPEND gs_ppend TO gt_ppend .

  ENDSELECT.

*--------------------------------------------------------------------*


**********************************************************************
*  SELECT wh_id, sku_id, batch,SUM( real_qty ) AS real_qty, end_date
*     FROM ztc_pp_end_list
*    INTO CORRESPONDING FIELDS OF TABLE @gt_ppend
*    GROUP BY wh_id, sku_id, batch, end_date .
**********************************************************************



  SORT gt_ppend BY erdat ASCENDING sku_id ASCENDING .

  LOOP AT gt_ppend INTO gs_ppend.
    IF gs_ppend-end_date IS INITIAL.
      CONTINUE.

    ELSE.

      READ TABLE lt_pp_year INTO ls_pp_year WITH KEY sku_id = gs_ppend-sku_id erdat = gs_ppend-end_date  .

      IF sy-subrc = 4.
*      **********************************************************************
*  case initial line.
*      **********************************************************************

        ls_pp_year-unit = 'EA'.

        ADD gs_ppend-real_qty TO ls_pp_year-total_fix_qty.

        READ TABLE gt_sku INTO ls_sku  WITH KEY sku_id = gs_ppend-sku_id .
        MOVE ls_sku-sku_id TO ls_pp_year-sku_id  .
        MOVE ls_sku-sku_name TO ls_pp_year-sku_name .

        CLEAR ls_sku.

        MOVE gs_ppend-end_date TO ls_pp_year-erdat .


        ls_pp_year-dom_unit = 'EA'.
        ls_pp_year-int_unit = 'EA'.

        IF gs_ppend-wh_id = '0000002000'.
          ADD gs_ppend-real_qty TO ls_pp_year-dom_qty.
        ELSEIF  gs_ppend-wh_id = '0000003000'.
          ADD gs_ppend-real_qty TO ls_pp_year-int_qty.


        ENDIF.


        ls_pp_year-dom_per = ( ls_pp_year-dom_qty / ls_pp_year-total_fix_qty ) * 100 .
        ls_pp_year-int_per = ( ls_pp_year-int_qty / ls_pp_year-total_fix_qty ) * 100 .
        APPEND ls_pp_year TO lt_pp_year.


*  modify lt_pp_year from ls_pp_year TRANSPORTING batch dom_per dom_qty dom_unit int_per int_qty int_unit sku_id sku_name total_fix_qty
*  where sku_id = ls_pp_year-sku_id and batch = ls_pp_year-batch and erdat = ls_pp_year-e


*        **********************************************************************
* case exist line.
      ELSE.
*        **********************************************************************

        ADD gs_ppend-real_qty TO ls_pp_year-total_fix_qty.

        IF gs_ppend-wh_id = '0000002000'.
          ADD gs_ppend-real_qty TO ls_pp_year-dom_qty.
        ELSEIF  gs_ppend-wh_id = '0000003000'.
          ADD gs_ppend-real_qty TO ls_pp_year-int_qty.
        ENDIF.

        ls_pp_year-dom_per = ( ls_pp_year-dom_qty / ls_pp_year-total_fix_qty ) * 100 .
        ls_pp_year-int_per = ( ls_pp_year-int_qty / ls_pp_year-total_fix_qty ) * 100 .

        MODIFY lt_pp_year FROM ls_pp_year
         TRANSPORTING dom_per int_per dom_qty int_qty total_fix_qty
         WHERE sku_id = ls_pp_year-sku_id .
      ENDIF.


      CLEAR : ls_pp_year.

    ENDIF.
  ENDLOOP.
**********************************************************************************************************************************************
  SORT lt_pp_year BY erdat(6) ASCENDING sku_id ASCENDING erdat+6(2) ASCENDING  .

**********************************************************************

*

*                              ''

                              USING

    ENDIF.

    PERFORM add_complete_line USING

                CHANGING pv_month.

*&      <-- LV_SKU_KEY

FORM add_sku_line TABLES pt_pp_year like gt_pp_year
                   USING
                            pv_relat_key TYPE lvc_nkey

                   CHANGING ps_pp_year TYPE zsc_pp_year
                     pv_sku_key.

  DATA : lv_node_text TYPE lvc_value,
        ls_pp_year LIKE LINE OF pt_pp_year,
         ls_line      TYPE zsc_pp_year.

LOOP AT pt_pp_year into ls_pp_year.
IF ls_pp_year-erdat(6) = ps_pp_year-erdat(6) and
   ls_pp_year-sku_id = ps_pp_year-sku_id.
add ls_pp_year-dom_qty to ls_line-dom_qty.
add ls_pp_year-int_qty to ls_line-int_qty.
ls_line-total_fix_qty = ls_line-dom_qty + ls_line-int_qty.
ENDIF.
ENDLOOP.

ls_line-iper_unit = '%'.
ls_line-dper_unit = '%'.

ls_line-dom_per = ( ls_line-dom_qty / ls_line-total_fix_qty ) * 100 .
ls_line-int_per = ( ls_line-int_qty / ls_line-total_fix_qty ) * 100 .

  lv_node_text = ps_pp_year-sku_id.

  CALL METHOD g_alv_tree->add_node
    EXPORTING
      i_relat_node_key = pv_relat_key

    IMPORTING

FORM mod_higharchy_erdat .
  DATA : lv_last_month like gs_pp_year-erdat(6),
    lv_index  TYPE i,
    lv_index2 TYPE i,
    lv_index_sku TYPE i,
    lv_index_sku2 TYPE i,
    lt_pyear  LIKE gt_pp_year,
    ls_pyear  LIKE LINE OF lt_pyear.

  DATA : lv_dom_qty LIKE gs_pp_year-dom_qty,
         lv_int_qty LIKE gs_pp_year-int_qty,
         lv_tot_qty LIKE gs_pp_year-total_fix_qty.

*********************************************************************

  LOOP AT gt_pp_year INTO gs_pp_year.
    IF lv_index IS INITIAL.
      lv_index = sy-tabix.

    ELSEIF gs_pp_year-erdat(6) IS INITIAL and
      gs_pp_year-dom_unit is not INITIAL.


      LOOP AT lt_pyear INTO ls_pyear.
        ADD ls_pyear-dom_qty TO lv_dom_qty .
        ADD ls_pyear-int_qty TO lv_int_qty.
        lv_tot_qty = lv_dom_qty + lv_int_qty.

      ENDLOOP.

      lv_index2 = sy-tabix.

      READ TABLE gt_pp_year INTO ls_pyear INDEX lv_index.

ls_pyear-dper_unit = '%'.
ls_pyear-iper_unit = '%'.
      ls_pyear-dom_per = ( lv_dom_qty / lv_tot_qty ) * 100 .
      ls_pyear-int_per = ( lv_int_qty / lv_tot_qty ) * 100 .

      MODIFY gt_pp_year FROM ls_pyear INDEX lv_index.
      lv_index =  lv_index2   .

      CLEAR : lt_pyear, ls_pyear, lv_int_qty, lv_tot_qty, lv_dom_qty.

    ELSE.

      APPEND gs_pp_year TO lt_pyear.

    ENDIF.



  ENDLOOP.

  LOOP AT lt_pyear INTO ls_pyear.
    ADD ls_pyear-dom_qty TO lv_dom_qty .
    ADD ls_pyear-int_qty TO lv_int_qty.
    lv_tot_qty = lv_dom_qty + lv_int_qty.

  ENDLOOP.
  READ TABLE gt_pp_year INTO ls_pyear INDEX lv_index.

ls_pyear-dper_unit = '%'.
ls_pyear-iper_unit = '%'.

  ls_pyear-dom_per = ( lv_dom_qty / lv_tot_qty ) * 100 .
  ls_pyear-int_per = ( lv_int_qty / lv_tot_qty ) * 100 .

  MODIFY gt_pp_year FROM ls_pyear INDEX lv_index.
  CLEAR : lt_pyear, ls_pyear, lv_int_qty, lv_tot_qty, lv_dom_qty.

*********************************************************************

ENDFORM.
