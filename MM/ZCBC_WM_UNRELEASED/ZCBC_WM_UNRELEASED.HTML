FORM get_data .
  DATA : ls_sku LIKE LINE OF gt_sku,
         ls_bp  LIKE LINE OF gt_bp.
DATA : ls_out_h TYPE ztc_wm_out_h.


*  SELECT  h~bp_id , i~sku_id , i~batch ,
*     sum( i~ord_qty ) as ord_qty, sum( i~out_qty ) as out_qty, sum( i~total_price ) as total_price
*    INTO CORRESPONDING FIELDS OF @gs_data
*    FROM ztc_sd_ord_h AS h
*    JOIN ztc_sd_ord_i AS i
*    ON h~ord_doc_num = i~ord_doc_num
*    where h~bp_id in @so_bp and
*          i~sku_id in @so_inv
*    GROUP BY  h~bp_id , i~sku_id , i~batch
*.


******* 뷰 연결 ******************



  SELECT BP_ID , SKU_ID, BATCH,
    SUM( ORD_QTY ) AS ORD_QTY , SUM( OUT_QTY ) AS OUT_QTY , SUM( TOTAL_PRICE ) AS TOTAL_PRICE
    INTO CORRESPONDING FIELDS OF @GS_DATA
    FROM ZVC_SD_ORD_DBV
    WHERE BP_ID IN @SO_BP
    AND  SKU_ID IN @SO_INV
    GROUP BY BP_ID , SKU_ID, BATCH.


      READ TABLE gt_sku INTO ls_sku WITH KEY sku_id = gs_data-sku_id .
      MOVE ls_sku-sku_name TO gs_data-sku_name.
      clear : ls_sku.

      READ TABLE gt_bp INTO ls_bp WITH KEY bp_id = gs_data-bp_id.
      MOVE ls_bp-bp_name TO gs_data-bp_name.
      clear : ls_bp.



*
*BREAK-POINT.

gs_data-percent = ( gs_data-out_qty / gs_data-ord_qty ) * 100.

*
      IF gs_data-percent < 50.
        gs_scol-fname = 'PERCENT'.
        gs_scol-color-col = 6.
        gs_scol-color-int = 1.
        gs_scol-color-inv = 0.
        APPEND gs_scol TO gs_data-itab_scol.
        clear gs_scol.
        ELSEIF gs_data-percent > 50 and gs_data-percent < 100.

        gs_scol-color-col = 7.
        gs_scol-color-int = 1.
        gs_scol-color-inv = 0.
        APPEND gs_scol TO gs_data-itab_scol.
        clear gs_scol.

        gs_scol-fname = 'PERCENT'.
        gs_scol-color-col = 5.
        gs_scol-color-int = 1.
        gs_scol-color-inv = 0.
        APPEND gs_scol TO gs_data-itab_scol.
        clear gs_scol.


      ENDIF.

      APPEND gs_data to  gt_data .
        clear gs_data.


ENDSELECT.
SORT gt_data by bp_id ASCENDING sku_id ASCENDING batch ASCENDING.



ENDFORM.
